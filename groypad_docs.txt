This guide contains everything trading terminals, bots, and aggregators need to integrate with Groypad — a bonding-curve token launchpad on TON. Groypad uses Uranus-compatible smart contracts, so if you already support Uranus trading, integration is straightforward.

Network: TON Mainnet
Graduation: 1,050 TON
Trade Fee: 3%

📈 Bonding Curve Math
Groypad uses a linear bonding curve: price(s) = α + β × s where s is the current supply.

Buy Quote (TON → tokens)
Solve for dx in the integral equation. Newton's method recommended for bigint precision.


const PRECISION = BigInt(1e9);

// Integral of linear curve from s1 to s2
function integrateCurve(s1: bigint, s2: bigint, alpha: bigint, beta: bigint): bigint {
  const dx = s2 - s1;
  if (dx <= 0n) return 0n;
  const term1 = (alpha * dx) / PRECISION;
  const term2 = (beta * dx * (s1 + s2)) / (2n * PRECISION * PRECISION);
  return term1 + term2;
}

// How many tokens for X nanoTON?
function buyQuote(amountTon: bigint, currentSupply: bigint, alpha: bigint, beta: bigint): bigint {
  const avgPrice = alpha + (beta * currentSupply) / PRECISION;
  if (avgPrice <= 0n) return 0n;
  let tokensOut = (amountTon * PRECISION) / avgPrice;

  // Newton's iterations
  for (let i = 0; i < 10; i++) {
    const cost = integrateCurve(currentSupply, currentSupply + tokensOut, alpha, beta);
    const diff = cost - amountTon;
    if (diff === 0n) break;
    const priceAtEnd = alpha + (beta * (currentSupply + tokensOut)) / PRECISION;
    if (priceAtEnd <= 0n) break;
    const adj = (diff * PRECISION) / priceAtEnd;
    tokensOut -= adj;
    if (tokensOut <= 0n) return 0n;
    if (adj > -2n && adj < 2n) break;
  }
  return tokensOut;
}
Sell Quote (tokens → TON)

function sellQuote(
  tokenAmount: bigint,
  currentSupply: bigint,
  alpha: bigint,
  beta: bigint,
  tradeFeeBPS: number = 100 // 1%
): bigint {
  if (tokenAmount <= 0n || tokenAmount > currentSupply) return 0n;
  const newSupply = currentSupply - tokenAmount;
  const rawTon = integrateCurve(newSupply, currentSupply, alpha, beta);
  // Contract deducts trading fee
  return (rawTon * (10000n - BigInt(tradeFeeBPS))) / 10000n;
}

  value: amountTON + 0.3 TON, // Buy amount + gas

⚠️ Gas: Always attach at least 0.3 TON for gas on top of the buy amount. Insufficient gas causes bounced transactions.


🔴 Sell Tokens
Selling uses the jetton burn mechanism. The message is sent to the user's MemeWallet (not the Meme contract).

Step 1: Resolve user's MemeWallet via get_wallet_address(owner) on the Meme contract.

Step 2: Send burn message to that MemeWallet address.

Message Body

// Sell/Burn message (TL-B):
// op:uint32 = 0x595f07bc
// query_id:uint64 = 0
// amount:coins             (tokens to sell)
// response_destination:address (owner wallet — receives TON proceeds)
// custom_payload:bit = 0

beginCell()
  .storeUint(0x595f07bc, 32)        // burn opcode
  .storeUint(0, 64)                  // query_id
  .storeCoins(tokenAmount)           // amount to sell
  .storeAddress(ownerAddress)        // receives TON
  .storeBit(false)                   // no custom payload
  .endCell()
Transaction Parameters

{
  to: "<MEME_WALLET_ADDRESS>",  // User's jetton wallet (NOT the meme master)
  value: 0.3 TON,               // Gas only — no extra TON needed
  body: sellMessageCell
}
⚠️ Important: Cap sell amount to the actual on-chain balance (from get_wallet_data). Rounding errors cause exit_code 27 bounces.


🎓 Graduation
When a token raises 1,050 TON, it graduates. The contract sets isGraduated = true and liquidity migrates to a DEX. After migration, isMigrated = true — the bonding curve is closed and trading happens on the DEX.


// Progress calculation
const GRADUATION_TARGET = 1050n * 1_000_000_000n; // 1050 TON in nanoTON
const progress = Math.min(100,
  Number((raisedFunds * 10000n) / GRADUATION_TARGET) / 100
);

// Status determination
if (isMigrated) status = 'graduated';      // Trade on DEX
else if (isGraduated) status = 'graduating'; // Liquidity migrating
else status = 'live';                        // Trade on bonding curve

🛠️ Full Integration Examples
TypeScript (@ton/ton)
Complete buy flow

import { TonClient, Address, beginCell, toNano } from '@ton/ton';

const client = new TonClient({ endpoint: 'https://toncenter.com/api/v2/jsonRPC' });

async function buyGroypadToken(
  memeAddress: string,
  amountTon: number,
  slippagePct: number = 5
) {
  const meme = Address.parse(memeAddress);
  
  // 1. Read bonding curve state
  const result = await client.runMethod(meme, 'get_meme_data');
  const stack = result.stack;
  stack.skip(7); // skip to alpha
  const alpha = stack.readBigNumber();
  const beta = stack.readBigNumber();
  stack.skip(1); // onSellSupply
  const tradeFeeBPS = stack.readNumber();
  stack.skip(1); // raisedFunds
  const currentSupply = stack.readBigNumber();

  // 2. Calculate expected tokens out
  const amountNano = toNano(amountTon.toString());
  const tokensOut = buyQuote(amountNano, currentSupply, alpha, beta);
  const minOut = (tokensOut * BigInt(100 - slippagePct)) / 100n;

  // 3. Build buy message
  const body = beginCell()
    .storeUint(0x742b36d8, 32)
    .storeUint(0, 64)
    .storeCoins(minOut)
    .endCell();

  // 4. Send transaction (value = buy amount + 0.3 TON gas)
  // Use TonConnect, ton-core WalletContractV4, or your preferred method
  return {
    to: meme.toString(),
    value: (amountNano + toNano('0.3')).toString(),
    body,
  };
}
Complete sell flow

async function sellGroypadToken(
  memeAddress: string,
  ownerAddress: string,
  tokenAmount: bigint
) {
  const meme = Address.parse(memeAddress);
  const owner = Address.parse(ownerAddress);

  // 1. Get user's MemeWallet address
  const walletResult = await client.runMethod(meme, 'get_wallet_address', [
    { type: 'slice', cell: beginCell().storeAddress(owner).endCell() },
  ]);
  const memeWallet = walletResult.stack.readAddress();

  // 2. Verify actual balance (prevent exit_code 27)
  const balResult = await client.runMethod(memeWallet, 'get_wallet_data');
  const actualBalance = balResult.stack.readBigNumber();
  const sellAmount = tokenAmount > actualBalance ? actualBalance : tokenAmount;

  // 3. Build burn/sell message
  const body = beginCell()
    .storeUint(0x595f07bc, 32)
    .storeUint(0, 64)
    .storeCoins(sellAmount)
    .storeAddress(owner)
    .storeBit(false)
    .endCell();

  // 4. Send to MemeWallet (not Meme contract!)
  return {
    to: memeWallet.toString(),
    value: toNano('0.3').toString(),
    body,
  };
}
Python (tonsdk / toncenter)
Bonding curve math

PRECISION = 10**9

def integrate_curve(s1: int, s2: int, alpha: int, beta: int) -> int:
    dx = s2 - s1
    if dx <= 0:
        return 0
    term1 = (alpha * dx) // PRECISION
    term2 = (beta * dx * (s1 + s2)) // (2 * PRECISION * PRECISION)
    return term1 + term2

def buy_quote(amount_ton: int, supply: int, alpha: int, beta: int) -> int:
    avg_price = alpha + (beta * supply) // PRECISION
    if avg_price <= 0:
        return 0
    tokens = (amount_ton * PRECISION) // avg_price
    for _ in range(10):
        cost = integrate_curve(supply, supply + tokens, alpha, beta)
        diff = cost - amount_ton
        if diff == 0:
            break
        p_end = alpha + (beta * (supply + tokens)) // PRECISION
        if p_end <= 0:
            break
        adj = (diff * PRECISION) // p_end
        tokens -= adj
        if tokens <= 0:
            return 0
        if -2 < adj < 2:
            break
    return tokens

def sell_quote(token_amount: int, supply: int, alpha: int, beta: int, fee_bps: int = 100) -> int:
    if token_amount <= 0 or token_amount > supply:
        return 0
    raw = integrate_curve(supply - token_amount, supply, alpha, beta)
    return (raw * (10000 - fee_bps)) // 10000

Full buy example

def prepare_buy(meme_address: str, ton_amount: float, slippage_pct: int = 5):
    """Returns dict with to, value, body for the buy transaction."""
    data = get_meme_data(meme_address)
    if data["migrated"] or data["is_graduated"]:
        raise Exception("Token already graduated — trade on DEX")

    amount_nano = int(ton_amount * 1e9)
    tokens = buy_quote(amount_nano, data["current_supply"], data["alpha"], data["beta"])
    min_out = (tokens * (100 - slippage_pct)) // 100

    body = build_buy_body(min_out)
    return {
        "to": meme_address,
        "value": amount_nano + 300_000_000,  # + 0.3 TON gas
        "body": body.to_boc(),
    }

  if (dx <= 0n) return 0n;

  const term2 = (beta * dx * (s1 + s2)) / (2n * PRECISION * PRECISION);
